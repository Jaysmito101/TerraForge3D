25static void zip_walk(struct zip_t *zip, const char *path, bool isFristLayer = true, std::string prevPath = "")
26{
27 DIR *dir;
28 struct dirent *entry;
29 char fullpath[MAX_PATH];
30 struct stat s;
31 memset(fullpath, 0, MAX_PATH);
32 dir = opendir(path);
33 assert(dir);
34
35 while ((entry = readdir(dir)))
36 {
37 // skip "." and ".."
38 if (!strcmp(entry->d_name, ".\0") || !strcmp(entry->d_name, "..\0"))
39 {
40 continue;
41 }
42
43 snprintf(fullpath, sizeof(fullpath), "%s/%s", path, entry->d_name);
44 stat(fullpath, &s);
45
46 if (S_ISDIR(s.st_mode))
47 {
48 zip_walk(zip, fullpath, false, prevPath + entry->d_name + "\\");
49 }
50
51 else
52 {
53 zip_entry_open(zip, (prevPath + entry->d_name).c_str());
54 zip_entry_fwrite(zip, fullpath);
55 zip_entry_close(zip);
56 }
57 }
58
59 closedir(dir);
60}
