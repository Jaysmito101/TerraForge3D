428void RegSet(HKEY hkeyHive, const char *pszVar, const char *pszValue)
429{
430 HKEY hkey;
431 char szValueCurrent[1000];
432 DWORD dwType;
433 DWORD dwSize = sizeof(szValueCurrent);
434 int iRC = RegGetValue(hkeyHive, CA2CT(pszVar), NULL, RRF_RT_ANY, &dwType, szValueCurrent, &dwSize);
435 bool bDidntExist = iRC == ERROR_FILE_NOT_FOUND;
436
437 if (iRC != ERROR_SUCCESS && !bDidntExist)
438 {
439 Log("RegGetValue( " + std::string(pszVar) + " ) Failed : " + strerror(iRC));
440 }
441
442 if (!bDidntExist)
443 {
444 if (dwType != REG_SZ)
445 {
446 Log("RegGetValue( " + std::string(pszVar) + " ) found type unhandled " + std::to_string(dwType));
447 }
448
449 if (strcmp(szValueCurrent, pszValue) == 0)
450 {
451 Log("RegSet( \"" + std::string(pszVar) + "\" \"" + std::string(pszValue) + "\" ): already correct");
452 return;
453 }
454 }
455
456 DWORD dwDisposition;
457 iRC = RegCreateKeyEx(hkeyHive, CA2CT(pszVar), 0, 0, 0, KEY_ALL_ACCESS, NULL, &hkey, &dwDisposition);
458
459 if (iRC != ERROR_SUCCESS)
460 {
461 Log("RegSetValue( " + std::string(pszVar) + " ) Failed : " + strerror(iRC));
462 }
463
464 iRC = RegSetValueEx(hkey, L"", 0, REG_SZ, (BYTE *)pszValue, strlen(pszValue) + 1);
465
466 if (iRC != ERROR_SUCCESS)
467 {
468 Log("RegSetValue( " + std::string(pszVar) + " ) Failed: " + strerror(iRC));
469 }
470
471 if (bDidntExist)
472 {
473 Log("RegSet( " + std::string(pszVar) +" ): set to \"" + std::string(pszValue) + "\"");
474 }
475
476 else
477 {
478 Log("RegSet( " + std::string(pszVar) +" ): changed \"" + std::string(szValueCurrent) + "\" to \"" + std::string(pszValue) + "\"");
479 }
480
481 RegCloseKey(hkey);
482}
