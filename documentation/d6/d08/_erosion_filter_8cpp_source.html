47HeightAndGradient CalculateHeightAndGradient(float *nodes, int mapSize, float posX, float posY)
48{
49 int coordX = (int)posX;
50 int coordY = (int)posY;
51 // Calculate droplet's offset inside the cell (0,0) = at NW node, (1,1) = at SE node
52 float x = posX - coordX;
53 float y = posY - coordY;
54 // Calculate heights of the four nodes of the droplet's cell
55 int nodeIndexNW = coordY * mapSize + coordX;
56 float heightNW = nodes[nodeIndexNW];
57 float heightNE = nodes[nodeIndexNW + 1];
58 float heightSW = nodes[nodeIndexNW + mapSize];
59 float heightSE = nodes[nodeIndexNW + mapSize + 1];
60 // Calculate droplet's direction of flow with bilinear interpolation of height difference along the edges
61 float gradientX = (heightNE - heightNW) * (1 - y) + (heightSE - heightSW) * y;
62 float gradientY = (heightSW - heightNW) * (1 - x) + (heightSE - heightNE) * x;
63 // Calculate height with bilinear interpolation of the heights of the nodes of the cell
64 float height = heightNW * (1 - x) * (1 - y) + heightNE * x * (1 - y) + heightSW * (1 - x) * y + heightSE * x * y;
65 return HeightAndGradient( height, gradientX, gradientY);
66}

102 {
103 int centreX = i % mapSize;
104 int centreY = i / mapSize;
105

107 {

109 addIndex = 0;
110
111 for (int y = -radius; y <= radius; y++)
112 {
113 for (int x = -radius; x <= radius; x++)
114 {
115 float sqrDst = x * x + y * y;
116
117 if (sqrDst < radius * radius)
118 {
119 int coordX = centreX + x;
120 int coordY = centreY + y;
121
122 if (coordX >= 0 && coordX < mapSize && coordY >= 0 && coordY < mapSize)
123 {
124 float weight = 1 - sqrt(sqrDst) / radius;
125 weightSum += weight;
126 weights[addIndex] = weight;
127 xOffsets[addIndex] = x;
128 yOffsets[addIndex] = y;
129 addIndex++;
130 }
131 }
132 }
133 }
134 }
135

144 erosionBrushWeights[i][j] = weights[j] / weightSum;
145 }
146 }

148 // END
149 float *map = new float[model->mesh->vertexCount];
150 float minHeightT = 100000000000000;
151 float maxHeightT = -100000000000000;
152 for (int i = 0; i < model->mesh->vertexCount; i++)
153 {
154 if(model->mesh->vert[i].position.y > maxHeightT)
155 maxHeightT = model->mesh->vert[i].position.y;
156 if(model->mesh->vert[i].position.y < minHeightT)
157 minHeightT = model->mesh->vert[i].position.y;
158 }
159 for (int i = 0; i < model->mesh->vertexCount; i++)

161 map[i] = (model->mesh->vert[i].position.y - minHeightT) / (maxHeightT - minHeightT);
162 }

165 currentMapSize = mapSize;
166 std::cout << std::endl << "Starting processing erosion." << std::endl;

168 for (int iteration = 0; iteration < numIterations; iteration++)
169 {
170 // Create water droplet at random point on map
171 float posX = (int)(((double)rand() / (RAND_MAX)) * mapSize-1);
172 float posY = (int)(((double)rand() / (RAND_MAX)) * mapSize - 1);
173
174 if (posX < 2)
175 {
176 posX = 2;
177 }
178
179 if (posY < 2)
180 {
181 posY = 2;
182 }
183
184 float dirX = 0;
185 float dirY = 0;
186 float speed = initialSpeed;
187 float water = initialWaterVolume;
188 float sediment = 0;
189
190 for (int lifetime = 0; lifetime < maxDropletLifetime; lifetime++)
191 {
192 int nodeX = (int)posX;
193 int nodeY = (int)posY;
194 int dropletIndex = nodeY * mapSize + nodeX;
195 // Calculate droplet's offset inside the cell (0,0) = at NW node, (1,1) = at SE node
196 float cellOffsetX = posX - nodeX;
197 float cellOffsetY = posY - nodeY;
198 // Calculate droplet's height and direction of flow with bilinear interpolation of surrounding heights
199 HeightAndGradient heightAndGradient = CalculateHeightAndGradient(map, mapSize, posX, posY);
200 // Update the droplet's direction and position (move position 1 unit regardless of speed)
201 dirX = (dirX * inertia - heightAndGradient.gradientX * (1 - inertia));
202 dirY = (dirY * inertia - heightAndGradient.gradientY * (1 - inertia));
203 // Normalize direction
204 float len = sqrt(dirX * dirX + dirY * dirY);
205
206 if (len != 0)
207 {
208 dirX /= len;
209 dirY /= len;
210 }
211
212 posX += dirX;
213 posY += dirY;
214
215 // Stop simulating droplet if it's not moving or has flowed over edge of map
216 if ((dirX == 0 && dirY == 0) || posX < 0 || posX >= mapSize - 1 || posY < 0 || posY >= mapSize - 1)
217 {
218 break;
219 }
220
221 // Find the droplet's new height and calculate the deltaHeight
222 float newHeight = CalculateHeightAndGradient(map, mapSize, posX, posY).height;
223 float deltaHeight = newHeight - heightAndGradient.height;
224 // Calculate the droplet's sediment capacity (higher when moving fast down a slope and contains lots of water)
225 float sedimentCapacity = MAX(-deltaHeight * speed * water * sedimentCapacityFactor, minSedimentCapacity);
226
227 // If carrying more sediment than capacity, or if flowing uphill:
228 if (sediment > sedimentCapacity || deltaHeight > 0)
229 {
230 // If moving uphill (deltaHeight > 0) try fill up to the current height, otherwise deposit a fraction of the excess sediment
231 float amountToDeposit = (deltaHeight > 0) ? MIN(deltaHeight, sediment) : (sediment - sedimentCapacity) * depositSpeed;
232 sediment -= amountToDeposit;
233 // Add the sediment to the four nodes of the current cell using bilinear interpolation
234 // Deposition is not distributed over a radius (like erosion) so that it can fill small pits
235 map[dropletIndex] += amountToDeposit * (1 - cellOffsetX) * (1 - cellOffsetY);
236 map[dropletIndex + 1] += amountToDeposit * cellOffsetX * (1 - cellOffsetY);
237 map[dropletIndex + mapSize] += amountToDeposit * (1 - cellOffsetX) * cellOffsetY;
238 map[dropletIndex + mapSize + 1] += amountToDeposit * cellOffsetX * cellOffsetY;
239 }
240
241 else
242 {
243 // Erode a fraction of the droplet's current carry capacity.
244 // Clamp the erosion to the change in height so that it doesn't dig a hole in the terrain behind the droplet
245 float amountToErode = MIN((sedimentCapacity - sediment) * erodeSpeed, -deltaHeight);
246
247 // Use erosion brush to erode from all nodes inside the droplet's erosion radius
248 for (int brushPointIndex = 0; brushPointIndex < erosionBrushIndicesSizes[dropletIndex]; brushPointIndex++)
249 {
250 int nodeIndex = erosionBrushIndices[dropletIndex][brushPointIndex];
251 float weighedErodeAmount = amountToErode * erosionBrushWeights[dropletIndex][brushPointIndex];
252 float deltaSediment = (map[nodeIndex] < weighedErodeAmount) ? map[nodeIndex] : weighedErodeAmount;
253 map[nodeIndex] -= deltaSediment;
254 sediment += deltaSediment;
255 }
256 }
257
258 // Update droplet's speed and water content
259 speed = sqrt(speed * speed + deltaHeight * gravity);
260 water *= (1 - evaporateSpeed);
261 }
262
263 if(iteration % 100000 == 0)
264 {
265 std::cout << "Processed " + std::to_string(iteration) << " particles.\r";
266 }
267 }
268
269 std::cout << "Finished Terrain Erosion Simulation\n";

271 for (int i = 0; i < model->mesh->vertexCount; i++)
272 {
273 float newHeight = (map[i] * (maxHeightT - minHeightT)) + minHeightT;
274 model->mesh->vert[i].extras1.z = model->mesh->vert[i].position.y - newHeight;
275 model->mesh->vert[i].position.y = newHeight;
276 }
277
