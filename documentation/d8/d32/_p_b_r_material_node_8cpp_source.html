5static void LoadPBRFunctions(GLSLHandler* handler)
6{
7 GLSLFunction DistributionGGX("DistributionGGX", "vec3 N, vec3 H, float roughness", "float");
8 DistributionGGX.AddLine(GLSLLine("float a = roughness * roughness;"));
9 DistributionGGX.AddLine(GLSLLine("float a2 = a * a;"));
10 DistributionGGX.AddLine(GLSLLine("float NdotH = max(dot(N, H), 0.0f);"));
11 DistributionGGX.AddLine(GLSLLine("float NdotH2 = NdotH * NdotH;"));
12 DistributionGGX.AddLine(GLSLLine("float denom = NdotH2 * (a2 - 1.0) + 1.0;"));
13 DistributionGGX.AddLine(GLSLLine("return a2 / (PI * denom * denom);"));
14 handler->AddFunction(DistributionGGX);
15
16 GLSLFunction GeometrySchlickGGX("GeometrySchlickGGX", "float NdotV, float roughness", "float");
17 GeometrySchlickGGX.AddLine(GLSLLine("float r = (roughness + 1.0);"));
18 GeometrySchlickGGX.AddLine(GLSLLine("float k = (r*r) / 8.0;"));
19 GeometrySchlickGGX.AddLine(GLSLLine(""));
20 GeometrySchlickGGX.AddLine(GLSLLine("float nom = NdotV;"));
21 GeometrySchlickGGX.AddLine(GLSLLine("float denom = NdotV * (1.0 - k) + k;"));
22 GeometrySchlickGGX.AddLine(GLSLLine(""));
23 GeometrySchlickGGX.AddLine(GLSLLine("return nom / denom;"));
24 handler->AddFunction(GeometrySchlickGGX);
25
26 GLSLFunction GeometrySmith("GeometrySmith", "vec3 N, vec3 V, vec3 L, float roughness", "float");
27 GeometrySmith.AddLine(GLSLLine("float NdotV = max(dot(N, V), 0.0);"));
28 GeometrySmith.AddLine(GLSLLine("float NdotL = max(dot(N, L), 0.0);"));
29 GeometrySmith.AddLine(GLSLLine(""));
30 GeometrySmith.AddLine(GLSLLine("float ggx2 = GeometrySchlickGGX(NdotV, roughness);"));
31 GeometrySmith.AddLine(GLSLLine("float ggx1 = GeometrySchlickGGX(NdotL, roughness);"));
32 GeometrySmith.AddLine(GLSLLine(""));
33 GeometrySmith.AddLine(GLSLLine("return ggx1 * ggx2;"));
34 handler->AddFunction(GeometrySmith);
35
36 GLSLFunction frenselSchlick("fresnelSchlick", "float cosTheta, vec3 F0", "vec3");
37 frenselSchlick.AddLine(GLSLLine("return F0 + (1.0 - F0) * pow(clamp(1.0 - cosTheta, 0.0, 1.0), 5.0);"));
38 handler->AddFunction(frenselSchlick);
39}
