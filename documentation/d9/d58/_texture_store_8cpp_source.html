11inline char *strcasestr(const char *str, const char *pattern)
12{
13 size_t i;
14
15 if (!*pattern)
16 {
17 return (char *)str;
18 }
19
20 for (; *str; str++)
21 {
22 if (toupper((unsigned char)*str) == toupper((unsigned char)*pattern))
23 {
24 for (i = 1;; i++)
25 {
26 if (!pattern[i])
27 {
28 return (char *)str;
29 }
30
31 if (toupper((unsigned char)str[i]) != toupper((unsigned char)pattern[i]))
32 {
33 break;
34 }
35 }
36 }
37 }
38
39 return NULL;
40}

146void TextureStore::LoadTextureDatabase()
147{

170 item.roughness = downloadedTextureDatabaseJ[item.name]["roughness"];
171 item.metallic = downloadedTextureDatabaseJ[item.name]["metallic"];
172 item.ao = downloadedTextureDatabaseJ[item.name]["ao"];
173 item.arm = downloadedTextureDatabaseJ[item.name]["arm"];
174 item.baseDir = downloadedTextureDatabaseJ[item.name]["baseDir"];
175 downloadedTextureStoreItems.push_back(textureStoreItems.size());
176 }
177
178 textureStoreItems.push_back(item);
